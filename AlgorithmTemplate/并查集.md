# #并查集

````cpp
class DSU {
public:
    vector<int> fa;

    DSU(int n) {
        fa.resize(n);
        for (int i = 0; i < n; ++i) fa[i] = i;
    }

    int find(int x) {
        if (fa[x] == x) return x;
        return fa[x] = find(fa[x]);
    }

    void unite(int x, int y) {
        x = find(x);
        y = find(y);
        if (x == y) return;
        fa[y] = x;
    }

    bool same(int x, int y) {
        return find(x) == find(y);
    }
};
````

路径压缩

````cpp
class DSU {
public:
    vector<int> fa, rk;

    DSU(int n) {
        fa.resize(n);
        rk.resize(n, 0);
        for (int i = 0; i < n; ++i) fa[i] = i;
    }

    int find(int x) {
        if (fa[x] == x) return x;
        return fa[x] = find(fa[x]);
    }

    void unite(int x, int y) {
        x = find(x);
        y = find(y);
        if (x == y) return;

        if (rk[x] < rk[y]) {
            fa[x] = y;
        } else if (rk[x] > rk[y]) {
            fa[y] = x;
        } else {
            fa[y] = x;
            rk[x]++;
        }
    }

    bool same(int x, int y) {
        return find(x) == find(y);
    }
};
````
